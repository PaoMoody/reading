  <script src="words.js"></script>

  <script>

    let currentIndex = 0;


    function renderWord() {

      const container = document.getElementById('wordContainer');

      container.innerHTML = '';

      const wordObj = wordData[currentIndex];

      wordObj.parts.forEach(part => {

        const img = document.createElement('img');

        img.src = part;

        img.alt = part;

        img.className = 'letter-img';

        container.appendChild(img);

      });

    }


    function playSounds() {

      const wordObj = wordData[currentIndex];

      let i = 0;

      function playNext() {

        if (i < wordObj.parts.length) {

          const audioFile = wordObj.parts[i].replace('.png', '.mp3');

          const audio = new Audio(audioFile);

          audio.onended = playNext;

          audio.play();

          i++;

        }

      }

      playNext();

    }


    function nextWord() {

      currentIndex = (currentIndex + 1) % wordData.length;

      renderWord();

    }


    // Initialize

    renderWord();

  </script>
